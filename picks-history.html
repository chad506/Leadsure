// Fund position symbols — picks matching these get highlighted
const FUND_SYMBOLS = new Set(['AAOI','ACLS','ADBE','AEHR','AMKR','AMSC','APLD','ARM','BE','BMBL','BXP','CAT','CBRE','CHGG','CIEN','COHR','COPX','CORZ','CRDO','CRM','CRWV','CTSH','CWK','DELL','DUOL','EXPI','FCG','FN','FVRR','GLW','GPN','HPP','HUT','INTC','INTU','IREN','LITE','LZ','MDB','MRVL','MU','NBIS','NOW','NVDA','ONTO','PSFE','REMX','RIOT','SNDK','SPG','STX','TASK','TSM','TTD','U','UPWK','VRT','WDAY','WDC','WPP','WULF','Z']);

// ---- FORMATTERS (same as app.js) ----
const fmtCurrency = new Intl.NumberFormat('en-US', {
style: 'currency', currency: 'USD',
minimumFractionDigits: 2, maximumFractionDigits: 2
});
const fmtPnLCurrency = new Intl.NumberFormat('en-US', {
style: 'currency', currency: 'USD',
minimumFractionDigits: 2, maximumFractionDigits: 2,
signDisplay: 'always'
});
const fmtInt = new Intl.NumberFormat('en-US');
const fmtPct = new Intl.NumberFormat('en-US', {
style: 'percent',
minimumFractionDigits: 2, maximumFractionDigits: 2,
signDisplay: 'always'
});
const fmtMarketCap = (val) => {
if (!val || val === 0) return '\u2014';
if (val >= 1e12) return '$' + (val / 1e12).toFixed(1) + 'T';
if (val >= 1e9) return '$' + (val / 1e9).toFixed(1) + 'B';
if (val >= 1e6) return '$' + (val / 1e6).toFixed(0) + 'M';
return '$' + val.toFixed(0);
};

// ---- BUILD FLAT PICKS ARRAY ----
function buildAllPicks() {
const all = [];
MODEL_PICKS.models.forEach(model => {
model.picks.forEach(p => {
const entry = {
...p,
entryPrice: p.price,
livePrice: p.livePrice || null,
prevClose: p.prevClose || null,
dateChosen: p.dateChosen || '',
model: model.name,
modelIcon: model.icon,
cost: p.qty * (p.livePrice || p.price),
gainLoss: 0,
gainLossPct: 0,
dayChangePct: 0
};
// Compute P&L from embedded livePrice if available
if (entry.livePrice !== null) {
const isLong = entry.direction === 'Long';
entry.cost = entry.livePrice * entry.qty;
entry.gainLoss = isLong
? (entry.livePrice - entry.entryPrice) * entry.qty
: (entry.entryPrice - entry.livePrice) * entry.qty;
entry.gainLossPct = entry.entryPrice !== 0
? (isLong
? (entry.livePrice - entry.entryPrice) / entry.entryPrice
: (entry.entryPrice - entry.livePrice) / entry.entryPrice)
: 0;
// Compute Day % from embedded prevClose
if (entry.prevClose && entry.prevClose !== 0) {
const rawDayPct = (entry.livePrice - entry.prevClose) / entry.prevClose;
entry.dayChangePct = isLong ? rawDayPct : -rawDayPct;
}
}
all.push(entry);
});
});
return all;
}

// ---- READ PRICES FROM LOCALSTORAGE (written by positions page) ----
// Picks-history does NOT call Finnhub directly to avoid rate-limit conflicts.
// The positions page fetches prices and broadcasts via localStorage.
function readSharedPrices() {
try {
const cached = localStorage.getItem('finnhub_prices');
const ts = parseInt(localStorage.getItem('finnhub_prices_ts') || '0');
if (!cached) return false;

// Only use if less than 5 minutes old
const ageMs = Date.now() - ts;
if (ageMs > 5 * 60 * 1000) {
console.log('[PicksHistory] Cached prices are stale (' + Math.round(ageMs/1000) + 's old), using embedded');
return false;
}

const priceMap = JSON.parse(cached);
let updated = 0;

allPicks.forEach(p => {
const live = priceMap[p.symbol];
if (live && live.c > 0) {
p.livePrice = live.c;
p.prevClose = live.pc;
p.cost = live.c * p.qty;

const isLong = p.direction === 'Long';
p.gainLoss = isLong
? (live.c - p.entryPrice) * p.qty
: (p.entryPrice - live.c) * p.qty;
p.gainLossPct = p.entryPrice !== 0
? (isLong
? (live.c - p.entryPrice) / p.entryPrice
: (p.entryPrice - live.c) / p.entryPrice)
: 0;

const rawDayPct = live.pc !== 0
? (live.c - live.pc) / live.pc
: 0;
p.dayChangePct = isLong ? rawDayPct : -rawDayPct;
updated++;
}
});

console.log('[PicksHistory] Read ' + updated + ' prices from localStorage (age: ' + Math.round(ageMs/1000) + 's)');
return updated > 0;
} catch (e) {
console.warn('[PicksHistory] localStorage read failed:', e);
return false;
}
}

// ---- FALLBACK: Slow Finnhub fetch if localStorage is empty/stale ----
const FINNHUB_API_KEY = typeof SHARED_FINNHUB_KEY !== 'undefined' ? SHARED_FINNHUB_KEY : 'd6kqa11r01qmopd1net0d6kqa11r01qmopd1netg';
let finnhubFetched = false;
async function fetchPickPricesFallback() {
if (finnhubFetched || !FINNHUB_API_KEY) return;
finnhubFetched = true;

const uniqueSymbols = [...new Set(allPicks.map(p => p.symbol))];
console.log('[PicksHistory] Fallback: fetching ' + uniqueSymbols.length + ' symbols from Finnhub (slow)');

const priceMap = {};
for (let i = 0; i < uniqueSymbols.length; i++) {
try {
const res = await fetch(
'https://finnhub.io/api/v1/quote?symbol=' + uniqueSymbols[i] + '&token=' + FINNHUB_API_KEY
);
if (res.ok) {
const data = await res.json();
if (data.c > 0) {
priceMap[uniqueSymbols[i]] = { c: data.c, pc: data.pc };
}
}
} catch (e) { /* skip */ }
if (i < uniqueSymbols.length - 1) {
await new Promise(r => setTimeout(r, 2000));
}
}

let updated = 0;
allPicks.forEach(p => {
const live = priceMap[p.symbol];
if (live && live.c > 0) {
p.livePrice = live.c;
p.prevClose = live.pc;
p.cost = live.c * p.qty;
const isLong = p.direction === 'Long';
p.gainLoss = isLong
? (live.c - p.entryPrice) * p.qty
: (p.entryPrice - live.c) * p.qty;
p.gainLossPct = p.entryPrice !== 0
? (isLong ? (live.c - p.entryPrice) / p.entryPrice : (p.entryPrice - live.c) / p.entryPrice)
: 0;
const rawDayPct = live.pc !== 0 ? (live.c - live.pc) / live.pc : 0;
p.dayChangePct = isLong ? rawDayPct : -rawDayPct;
updated++;
}
});

// Save to localStorage
try {
const existing = JSON.parse(localStorage.getItem('finnhub_prices') || '{}');
Object.assign(existing, priceMap);
localStorage.setItem('finnhub_prices', JSON.stringify(existing));
localStorage.setItem('finnhub_prices_ts', Date.now().toString());
} catch (e) { /* skip */ }

console.log('[PicksHistory] Fallback fetched ' + updated + ' prices');
renderHistoryTable();
}

// Poll localStorage for fresh prices every 10s
function pollSharedPrices() {
if (readSharedPrices()) {
renderHistoryTable();
}
}

// ---- SORT STATE ----
let currentSort = { key: 'symbol', dir: 'asc' };
let allPicks = [];

// ---- RENDER TABLE (desktop) + CARDS (mobile) ----
function renderHistoryTable() {
// Desktop table
const tbody = document.getElementById('history-body');
const cardsEl = document.getElementById('history-cards');

const sorted = [...allPicks].sort((a, b) => {
let aVal = a[currentSort.key];
let bVal = b[currentSort.key];
if (typeof aVal === 'string') {
aVal = aVal.toLowerCase();
bVal = (bVal || '').toLowerCase();
}
if (aVal < bVal) return currentSort.dir === 'asc' ? -1 : 1;
if (aVal > bVal) return currentSort.dir === 'asc' ? 1 : -1;
return 0;
});

// Render desktop table rows
if (tbody) {
tbody.innerHTML = sorted.map(p => {
const dirClass = p.direction === 'Long' ? 'dir-long' : 'dir-short';
const dirLabel = p.direction === 'Long' ? 'LONG' : 'SHORT';
const rowClass = p.direction === 'Long' ? 'row-long' : 'row-short';
const inFund = FUND_SYMBOLS.has(p.symbol);
const fundClass = inFund ? ' row-in-fund' : '';
const priceDisplay = p.livePrice !== null ? fmtCurrency.format(p.livePrice) : '\u2014';
const pnlClass = p.gainLoss >= 0 ? 'num-positive' : 'num-negative';
const pnlDisplay = p.livePrice !== null ? fmtPnLCurrency.format(p.gainLoss) : '\u2014';
const pnlPctDisplay = p.livePrice !== null ? fmtPct.format(p.gainLossPct) : '\u2014';
const dayClass = p.dayChangePct >= 0 ? 'num-positive' : 'num-negative';
const dayDisplay = p.livePrice !== null ? fmtPct.format(p.dayChangePct) : '\u2014';
const fundBadge = inFund ? '<span class="fund-badge" title="Also a live fund position">LIVE</span>' : '';

return `
<tr class="${rowClass}${fundClass}">
<td class="symbol-cell sticky-col">${p.symbol} ${fundBadge}</td>
<td class="name-cell" title="${p.name}"><a href="https://www.marketwatch.com/investing/stock/${p.symbol.toLowerCase()}" target="_blank" rel="noopener noreferrer">${p.name}</a></td>
<td class="industry-cell" title="${p.industry || ''}">${p.industry || ''}</td>
<td class="mktcap-cell">${fmtMarketCap(p.marketCap)}</td>
<td class="model-cell"><span class="model-icon">${p.modelIcon}</span> ${p.model}</td>
<td class="date-cell">${p.dateChosen}</td>
<td><span class="dir-badge ${dirClass}">${dirLabel}</span></td>
<td class="num-cell">${fmtInt.format(p.qty)}</td>
<td class="num-cell">${fmtCurrency.format(p.entryPrice)}</td>
<td class="num-cell">${priceDisplay}</td>
<td class="num-cell">${fmtCurrency.format(p.cost)}</td>
<td class="num-cell ${p.livePrice !== null ? pnlClass : ''}">${pnlDisplay}</td>
<td class="num-cell ${p.livePrice !== null ? pnlClass : ''}">${pnlPctDisplay}</td>
<td class="num-cell ${p.livePrice !== null ? dayClass : ''}">${dayDisplay}</td>
<td class="rationale-cell" title="${p.rationale}">${p.rationale}</td>
</tr>
`;
}).join('');
}

// Render mobile cards
if (cardsEl) {
cardsEl.innerHTML = sorted.map(p => {
const dirClass = p.direction === 'Long' ? 'dir-long' : 'dir-short';
const dirLabel = p.direction === 'Long' ? 'LONG' : 'SHORT';
const borderClass = p.direction === 'Long' ? 'hcard-long' : 'hcard-short';
const inFund = FUND_SYMBOLS.has(p.symbol);
const fundCardClass = inFund ? ' hcard-in-fund' : '';
const priceDisplay = p.livePrice !== null ? fmtCurrency.format(p.livePrice) : '\u2014';
const pnlClass = p.gainLoss >= 0 ? 'num-positive' : 'num-negative';
const pnlDisplay = p.livePrice !== null ? fmtPnLCurrency.format(p.gainLoss) : '\u2014';
const pnlPctDisplay = p.livePrice !== null ? fmtPct.format(p.gainLossPct) : '\u2014';
const fundBadge = inFund ? '<span class="fund-badge" title="Also a live fund position">LIVE</span>' : '';

return `
<div class="hcard ${borderClass}${fundCardClass}">
<div class="hcard-top">
<div class="hcard-symbol-group">
<a class="hcard-symbol" href="https://www.marketwatch.com/investing/stock/${p.symbol.toLowerCase()}" target="_blank" rel="noopener noreferrer">${p.symbol}</a>
<span class="dir-badge ${dirClass}">${dirLabel}</span>
${fundBadge}
</div>
<span class="hcard-pnl ${p.livePrice !== null ? pnlClass : ''}">${pnlDisplay}</span>
</div>
<div class="hcard-name">${p.name}</div>
<div class="hcard-model"><span class="model-icon">${p.modelIcon}</span> ${p.model} &middot; ${p.dateChosen} &middot; ${p.industry || ''} &middot; ${fmtMarketCap(p.marketCap)}</div>
<div class="hcard-stats">
<div class="hcard-stat">
<span class="hcard-stat-label">Entry</span>
<span class="hcard-stat-value">${fmtCurrency.format(p.entryPrice)}</span>
</div>
<div class="hcard-stat">
<span class="hcard-stat-label">Price</span>
<span class="hcard-stat-value">${priceDisplay}</span>
</div>
<div class="hcard-stat">
<span class="hcard-stat-label">Qty</span>
<span class="hcard-stat-value">${fmtInt.format(p.qty)}</span>
</div>
<div class="hcard-stat">
<span class="hcard-stat-label">P&L %</span>

</div>
</div>

</div>

}
}

// ---- SORT HANDLER ----
function initSort() {
const headers = document.querySelectorAll('#history-table th[data-sort]');
headers.forEach(th => {
th.addEventListener('click', () => {
const key = th.dataset.sort;
if (currentSort.key === key) {
currentSort.dir = currentSort.dir === 'asc' ? 'desc' : 'asc';
} else {
currentSort.key = key;
currentSort.dir = 'asc';
}
headers.forEach(h => h.classList.remove('sort-asc', 'sort-desc'));
th.classList.add(currentSort.dir === 'asc' ? 'sort-asc' : 'sort-desc');
renderHistoryTable();
});
});
}

// ---- THEME TOGGLE ----
function initTheme() {
const toggle = document.querySelector('[data-theme-toggle]');
const root = document.documentElement;
let theme = root.getAttribute('data-theme') || 'dark';

if (toggle) {
toggle.innerHTML = theme === 'dark'
? '<svg width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><circle cx="12" cy="12" r="5"/><path d="M12 1v2M12 21v2M4.22 4.22l1.42 1.42M18.36 18.36l1.42 1.42M1 12h2M21 12h2M4.22 19.78l1.42-1.42M18.36 5.64l1.42-1.42"/></svg>'
: '<svg width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M21 12.79A9 9 0 1 1 11.21 3 7 7 0 0 0 21 12.79z"/></svg>';

toggle.addEventListener('click', () => {
theme = theme === 'dark' ? 'light' : 'dark';
root.setAttribute('data-theme', theme);
toggle.setAttribute('aria-label', `Switch to ${theme === 'dark' ? 'light' : 'dark'} mode`);
toggle.innerHTML = theme === 'dark'
? '<svg width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><circle cx="12" cy="12" r="5"/><path d="M12 1v2M12 21v2M4.22 4.22l1.42 1.42M18.36 18.36l1.42 1.42M1 12h2M21 12h2M4.22 19.78l1.42-1.42M18.36 5.64l1.42-1.42"/></svg>'
: '<svg width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M21 12.79A9 9 0 1 1 11.21 3 7 7 0 0 0 21 12.79z"/></svg>';
});
}
