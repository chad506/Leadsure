// ---- COMPUTE P&L FOR A POSITION ----
function computePnl(pos) {
const isLong = pos.direction === 'Long';
if (pos.status === 'sold') {
return { pnl: pos.realizedPnl, pnlPct: pos.costBasis !== 0 ? pos.realizedPnl / (pos.costBasis * pos.qty) : 0 };
}
const pnl = isLong
? (pos.currentPrice - pos.costBasis) * pos.qty
: (pos.costBasis - pos.currentPrice) * pos.qty;
const pnlPct = pos.costBasis !== 0
? (isLong
? (pos.currentPrice - pos.costBasis) / pos.costBasis
: (pos.costBasis - pos.currentPrice) / pos.costBasis)
: 0;
return { pnl, pnlPct };
}

// ---- RENDER SUMMARY KPIS ----
function renderKpis() {
const active = FUND_POSITIONS.filter(p => p.status === 'active');
const sold = FUND_POSITIONS.filter(p => p.status === 'sold');
const addons = FUND_POSITIONS.filter(p => p.status === 'addon');

// Realized P&L (from sold positions)
const realizedPnl = sold.reduce((sum, p) => sum + (p.realizedPnl || 0), 0);

// Unrealized P&L (from active positions only — not add-ons which are sub-lots of active)
const unrealizedPnl = active.reduce((sum, p) => {
const { pnl } = computePnl(p);
return sum + pnl;
}, 0);

const combinedPnl = realizedPnl + unrealizedPnl;

// Win rate across all closed + active
const allForWinRate = [...active, ...sold];
const winners = allForWinRate.filter(p => computePnl(p).pnl > 0).length;
const winRate = allForWinRate.length > 0 ? (winners / allForWinRate.length * 100).toFixed(1) : '0';

document.getElementById('fr-total-positions').textContent = active.length + sold.length;
document.getElementById('fr-active-count').textContent = active.length;

}

// ---- FILTER POSITIONS ----
function getFilteredPositions() {
return FUND_POSITIONS.filter(p => {
if (currentFilter === 'all') return true;
if (currentFilter === 'active') return p.status === 'active';
if (currentFilter === 'sold') return p.status === 'sold';
if (currentFilter === 'addon') return p.status === 'addon';
if (currentFilter === 'long') return p.direction === 'Long';
if (currentFilter === 'short') return p.direction === 'Short';
return true;
});
}

// ---- SORT POSITIONS ----
function sortPositions(positions) {
return [...positions].sort((a, b) => {
let aVal, bVal;
const key = currentSort.key;

if (key === 'pnl') {
aVal = computePnl(a).pnl;
bVal = computePnl(b).pnl;
} else if (key === 'pnlPct') {
aVal = computePnl(a).pnlPct;
bVal = computePnl(b).pnlPct;
} else if (key === 'notional') {
aVal = a.currentPrice * a.qty;
bVal = b.currentPrice * b.qty;
} else if (key === 'currentPrice') {
aVal = a.status === 'sold' ? (a.exitPrice || a.currentPrice) : a.currentPrice;
bVal = b.status === 'sold' ? (b.exitPrice || b.currentPrice) : b.currentPrice;
} else {
aVal = a[key];
bVal = b[key];
}

if (typeof aVal === 'string') {
aVal = aVal.toLowerCase();
bVal = (bVal || '').toLowerCase();
}
if (aVal < bVal) return currentSort.dir === 'asc' ? -1 : 1;
if (aVal > bVal) return currentSort.dir === 'asc' ? 1 : -1;
return 0;
});
}

// ---- READ PRICES FROM LOCALSTORAGE (written by positions page) ----
// Fund-results does NOT call Finnhub directly to avoid rate-limit conflicts.
// The positions page fetches prices and broadcasts via localStorage.
function readSharedPrices() {
try {
const cached = localStorage.getItem('finnhub_prices');
const ts = parseInt(localStorage.getItem('finnhub_prices_ts') || '0');
if (!cached) return false;

// Only use if less than 5 minutes old
const ageMs = Date.now() - ts;
if (ageMs > 5 * 60 * 1000) {
console.log('[FundResults] Cached prices are stale (' + Math.round(ageMs/1000) + 's old), using embedded');
return false;
}

const priceMap = JSON.parse(cached);
let updated = 0;

// Update FUND_POSITIONS with cached prices
FUND_POSITIONS.forEach(p => {
if (p.status !== 'sold' && priceMap[p.symbol]) {
p.currentPrice = priceMap[p.symbol].c;
updated++;
}
});

// Also update shared POSITIONS
POSITIONS.forEach(pos => {
if (priceMap[pos.symbol]) {
pos.price = priceMap[pos.symbol].c;
if (priceMap[pos.symbol].pc > 0) {
pos.prevClose = priceMap[pos.symbol].pc;
}
}
});

console.log('[FundResults] Read ' + updated + ' prices from localStorage (age: ' + Math.round(ageMs/1000) + 's)');
return updated > 0;
} catch (e) {
console.warn('[FundResults] localStorage read failed:', e);
return false;
}
}

// ---- FALLBACK: Slow Finnhub fetch if localStorage is empty/stale ----
let finnhubFetched = false;
async function fetchLivePricesFallback() {
if (finnhubFetched || !SHARED_FINNHUB_KEY) return;
finnhubFetched = true;

const activePositions = FUND_POSITIONS.filter(p => p.status !== 'sold');
const uniqueSymbols = [...new Set(activePositions.map(p => p.symbol))];
console.log('[FundResults] Fallback: fetching ' + uniqueSymbols.length + ' symbols from Finnhub (slow)');

// Fetch one at a time with 2s delay (~30 calls/min, leaves room for positions page)
const priceMap = {};
for (let i = 0; i < uniqueSymbols.length; i++) {
try {
const res = await fetch(
'https://finnhub.io/api/v1/quote?symbol=' + uniqueSymbols[i] + '&token=' + SHARED_FINNHUB_KEY
);
if (res.ok) {
const data = await res.json();
if (data.c > 0) {
priceMap[uniqueSymbols[i]] = { c: data.c, pc: data.pc };
}
}
} catch (e) { /* skip */ }
if (i < uniqueSymbols.length - 1) {
await new Promise(r => setTimeout(r, 2000));
}
}

let updated = 0;
FUND_POSITIONS.forEach(p => {
if (p.status !== 'sold' && priceMap[p.symbol]) {
p.currentPrice = priceMap[p.symbol].c;
updated++;
}
});
POSITIONS.forEach(pos => {
if (priceMap[pos.symbol]) {
pos.price = priceMap[pos.symbol].c;
if (priceMap[pos.symbol].pc > 0) pos.prevClose = priceMap[pos.symbol].pc;
}
});

// Save to localStorage for other pages
try {
const existing = JSON.parse(localStorage.getItem('finnhub_prices') || '{}');
Object.assign(existing, priceMap);
localStorage.setItem('finnhub_prices', JSON.stringify(existing));
localStorage.setItem('finnhub_prices_ts', Date.now().toString());
} catch (e) { /* skip */ }

console.log('[FundResults] Fallback fetched ' + updated + ' prices');
renderKpis();
renderTable();
}

// Poll localStorage for fresh prices every 10s
function pollSharedPrices() {
if (readSharedPrices()) {
renderKpis();
renderTable();
}
}

// ---- INIT SORT HANDLERS ----
function initSort() {
const headers = document.querySelectorAll('#fr-table th[data-sort]');
headers.forEach(th => {
th.addEventListener('click', () => {
const key = th.dataset.sort;
if (currentSort.key === key) {
currentSort.dir = currentSort.dir === 'asc' ? 'desc' : 'asc';
} else {
currentSort.key = key;
currentSort.dir = 'asc';
}
headers.forEach(h => h.classList.remove('sort-asc', 'sort-desc'));
th.classList.add(currentSort.dir === 'asc' ? 'sort-asc' : 'sort-desc');
renderTable();
});
});
}

btn.addEventListener('click', () => {
btns.forEach(b => b.classList.remove('active'));
btn.classList.add('active');
currentFilter = btn.dataset.filter;
renderTable();
});
});
}
